# ============================================================
# 使用步骤 (3步搞定)
# ============================================================

## 第一步: 安装依赖
pip install flask requests

## 第二步: 启动SAM3服务器 (终端1)
python sam3_http_server.py

# 或使用便捷脚本
python run_sam3_server.py

# 或后台启动
python sam3_http_server.py &

## 第三步: 运行训练脚本 (终端2)
python train_qwen_edit_lora_v402.py --config your_config.yaml

# ============================================================
# 工作流程
# ============================================================

1. SAM3服务器启动并加载模型（独立进程）
   ↓
2. 训练脚本运行，当需要计算IoU时...
   ↓
3. 通过HTTP向SAM3服务器发送请求
   ↓
4. 服务器处理并返回IoU结果
   ↓
5. 训练脚本继续使用该结果

## 常见问题
- "连接拒绝" → 服务器未启动
- "端口被占用" → 改用其他端口: python sam3_http_server.py --port 5001
- "超时" → 图像太大，减小分辨率

# ============================================================
# 在代码中自定义
# ============================================================

from sam3_client import Sam3HttpClient

# 自定义服务器地址
client = Sam3HttpClient("http://127.0.0.1:5001", timeout=600)

# 计算IoU
iou = client.process_images("img1.jpg", "img2.jpg", "object_name")

# 分割图像
mask, shape = client.segment("image.jpg", "object_name")

print(f"IoU: {iou:.4f}")
print(f"Mask shape: {shape}")
